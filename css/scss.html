Modules

//You can split into mutiple files and @use rule. 
$font-stack:    Helvetica, sans-serif;
$primary-color: #333;
body {
  font: 100% $font-stack;
  color: $primary-color;
}
---------
@use 'base';
.inverse {
  background-color: base.$primary-color;
  color: white;
}


Mixins

//Mixins - groups of CSS declarations that 
//you want to reuse throughout your site
@mixin transform($property) {
  -webkit-transform: $property;
  -ms-transform: $property;
  transform: $property;
}
.box { @include transform(rotate(30deg)); }


Extend/Inheritance

//Using @extend lets you share a set of 
//CSS properties from one selector to another.
%message-shared {
  border: 1px solid #ccc;
  padding: 10px;
  color: #333;
}
.success {
  @extend %message-shared;
  border-color: green;
}
