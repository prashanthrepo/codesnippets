Array - Print items inbetween

Array print numbers in between
const numbers = arrayFromRange(-10,4);
console.log(numbers);
function arrayFromRange(min,max){
    var a =[]
    for(var i = min; i <= max; i++){
            a.push(i)
    }
    return a;
}

  let c = [];
