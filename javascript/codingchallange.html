function callfizzbuzz(input) {
if(typeof input !== 'number')
    return 'Not a Number';
if(input % 3 === 0 && input % 5 === 0)
    return 'FizzBuzz'
if(input % 3 === 0) 
    return 'Fizz';
if(input % 5 === 0)
    return 'Buzz'
return input
}


Check if anagrams/ same names

var firstWord = "Mary";
var secondWord = "Army";
isAnagram(firstWord, secondWord);
function isAnagram(first, second) {
    var a = first.toLowerCase();
    var b = second.toLowerCase();
    a = a.split("").sort().join("");
    b = b.split("").sort().join("");
return a === b;}
