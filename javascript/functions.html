Constructor Function

function TrimString(string){
    this.string = string;
    this.do = function(){
        return this.string.trim()
    }
}
const a = new TrimString('    Hello    ');
a.do()

lodash - compose and pipe

import { compose, pipe } from "lodash/fp";

const input = "      Prashanth B R    ";
const trim = (str) => str.trim();
const wrapdiv = (str) => `
${str}
`;
const lowercase = (str) => str.toLowerCase();

const transform = pipe(trim, lowercase, wrapdiv);
const transform = compose(wrapdiv, lowercase, trim);
transform(input);
