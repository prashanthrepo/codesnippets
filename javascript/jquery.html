    $( this ).removeClass( "inside" );
  }

AJAX

//AJAX is the art of exchanging data with a server, and 
//updating parts of a web page - without reloading the whole page. 
//The load() method loads data from a server 
//and puts the returned data into the selected element. 
$("#div1").load("demo_test.txt"); 
$("button").click(function(){ 
   $("#div1").load("demo_test.txt", function(responseTxt, statusTxt, xhr){ 
     if(statusTxt == "success") 
       alert("External content loaded successfully!"); 
     if(statusTxt == "error") 
       alert("Error: " + xhr.status + ": " + xhr.statusText); 
   }); 
});

        //Ajax call to get products data
        function getProducts(){
        $.ajax({
        url: 'https://fakestoreapi.com/products',
        type:'GET',
        success: function(data){
        console.log(data)
        //const res = JSON.stringify(data)
        appendData(data)
        },
        error: function(e){
        console.log(e)
        }
        })
        }
        getProducts();
        
        //Append Product Data
        function appendData(data){
        //Filter data - price which is higher than 50
        const filteredData = data.filter(item => {
        return item.price > 50;
        })
        
        //Sort filtered data > ascending order
        const sortedData = filteredData.sort((a,b) => {
        return b.price - a.price;
        })
        
        //Find an item in sorted list
        const uniqueItem = sortedData.find(item => {
        return item.title.includes('2TB');
        })
        console.log(`uniqueItem`, uniqueItem)
        
        //Total of all the items
        let totalprice = 0;
        let allItems = []
        for(i of sortedData){
        totalprice += i.price;
        allItems.push(i.price)
        }
        $('#totalprice').text(totalprice.toFixed(2))
        
        //Total using reduce
        const totalReduce = allItems.reduce((a,b) => a + b)
        $('#totalpricereduce').text(totalReduce.toFixed(2))
        
        //Append filtered data
        let html = '';
        sortedData.forEach(elm => {
        html += `
  • ${elm.price}
    • `
        });
        document.getElementById('products').innerHTML = html;
        }
