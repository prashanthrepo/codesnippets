useState - reactive values (any datatype),

// State for the component
import React, { useState } from "react";
function app() {
  const [name, setName] = useState("Prashanth");
  handleClick = () => {
    setName("Srikanth");
  };
  return (
    <div>
      <p>{name}</p>
      <button onClick={handleClick}>Click</button>
    </div>
  );
}
export default app;



useEffect - runs on every render.

//useEffect runs on every render.
//Just like component did mount.

import {useEffect} from 'react'
let name = 'hello'
useEffect(() => {
    console.log('Hello')
}, []);

//Second empty parameter to run the useeffect only once.
[name] = > Runs on change of variable.
useEffect(() => {
    console.log('Hello')
}, [name]);



useRef

//useRef he useRef Hook allows us to create 
//mutable variables in functional components.

function TextInputWithFocusButton() {
  const inputEl = useRef(null);
  const onButtonClick = () => {
    // `current` points to the mounted text input element
    inputEl.current.focus();
  };
  return (
    <>
      <input ref={inputEl} type="text" />
      <button onClick={onButtonClick}>Focus the input</button>
    </>
  );
}



useParam - to access attributes of an url

import React, { useParams } from 'react-router-dom'
const Blogdetails = () => {
  const { id } = useParams()
  return (
    <div> className="blog-details">
      <h2> Blog details : {id} </h2>
    </div>
  )
}
export default Blogdetails;


useHistory - to access history

import React, { useHistory } from 'react-router-dom'
const goback = () => {
  const history = useHistory()
  history.go(-1)
  history.push('/home');
}
export default goback;
